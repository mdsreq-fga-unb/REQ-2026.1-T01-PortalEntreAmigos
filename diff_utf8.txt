+export function ConfirmacaoEmail() {
+  const navigate = useNavigate();
+  const [searchParams] = useSearchParams();
+  const uid = searchParams.get('uid');
+  const token = searchParams.get('token');
+
+  const [status, setStatus] = useState<Status>('loading');
+  const [errorMessage, setErrorMessage] = useState('');
+
+  // Evita disparar a confirma├º├úo duas vezes (StrictMode roda o effect 2x em dev)
+  const jaConfirmou = useRef(false);
+
+  useEffect(() => {
+    if (jaConfirmou.current) return;
+    jaConfirmou.current = true;
+
+    const confirmarEmail = async () => {
+      if (!uid || !token) {
+        setStatus('error');
+        setErrorMessage('Link de confirma├º├úo inv├ílido ou incompleto.');
+        return;
+      }
+
+      try {
+        // Ajuste a URL conforme o endpoint real do backend Django
+        const response = await fetch('http://localhost:8000/api/ativar-conta/', {
+          method: 'POST',
+          headers: { 'Content-Type': 'application/json' },
+          body: JSON.stringify({ uid, token }),
+        });
+
+        if (!response.ok) {
+          throw new Error('Este link j├í foi usado ou expirou.');
+        }
+
+        setStatus('success');
+      } catch (err: any) {
+        setStatus('error');
+        setErrorMessage(err.message || 'N├úo foi poss├¡vel confirmar seu e-mail.');

+    };
+
+    confirmarEmail();
